CT 는 PR, RX 필수

SELECT RXH.*
FROM MM_RX_H RXH
           INNER JOIN MM_RX_D2 RXS
                   ON RXS.RX_H_UID = RXH.RX_H_UID
                  AND RXS.ACTV_FLAG = 'Y'
           INNER JOIN MD_SUPPLIER SUP
                   ON SUP.SUPPLIER_CODE = RXS.SUPPLIER_CODE
                  AND SUP.ACTV_FLAG = 'Y'
               --  AND NVL (SUP.ERP_CODE, 'N') != 'N'
           INNER JOIN MM_BD_H BDH
                   ON BDH.RX_H_UID = BDH.RX_H_UID
                  AND BDH.RX_D2_UID = RXS.RX_D2_UID
                  AND BDH.ACTV_FLAG = 'Y'
           INNER JOIN MM_BD_D1 BDI
                   ON BDI.BD_H_UID = BDH.BD_H_UID
                  AND BDI.BD_D1_UID =
                      (SELECT MIN (BD_D1_UID)
                         FROM MM_BD_D1 B
                        WHERE B.BD_H_UID = BDH.BD_H_UID
                          AND B.SEL_FLAG = 'Y'
                      )
           INNER JOIN MM_RX_D1 RXI
                   ON RXI.RX_H_UID = RXH.RX_H_UID
                  AND RXI.RX_D1_UID = BDI.RX_D1_UID
                  AND RXI.ACTV_FLAG = 'Y'
        LEFT OUTER JOIN MM_PR_H PRH
                   ON PRH.PR_H_UID = RXI.PR_H_UID
                  AND PRH.ACTV_FLAG = 'Y'
         WHERE RXH.ACTV_FLAG = 'Y'
           AND RXH.STATUS = 'CFM'
           AND RXH.RFX_TYPE_CODE IN ('RQ', 'BD', 'RA')
           AND RXH.CONFIRM_CT_FLAG = 'Y'
           AND NOT EXISTS
               (SELECT 1
                  FROM MM_CT_H CTH
                 WHERE CTH.RX_H_UID = RXH.RX_H_UID
                   AND CTH.BD_H_UID = BDH.BD_H_UID
                   AND CTH.ACTV_FLAG = 'Y'
               )
          AND    RXH.RX_CONFIRM_DATE >= TO_DATE('2025-07-13', 'yyyy-MM-dd') AND    RXH.RX_CONFIRM_DATE <= TO_DATE('2025-08-13 23:59:59', 'yyyy-MM-dd hh24:mi:ss');


SELECT RX.STATUS, BD.STATUS, RX.RX_NO, RXS.SUPPLIER_NAME, RXS.SEL_FLAG , RXI.RX_LINE_NO, RXI.ITEM_CODE, BDI.SEL_FLAG , RX.SEL_METHOD_CODE
FROM MM_RX_H RX
INNER JOIN MM_RX_D1 RXI ON (RX.RX_H_UID = RXI.RX_H_UID AND RXI.ACTV_FLAG = 'Y')
INNER JOIN MM_RX_D2 RXS ON (RX.RX_H_UID = RXS.RX_H_UID AND RXS.ACTV_FLAG = 'Y')
INNER JOIN MM_BD_H BD ON (RXS.RX_H_UID = BD.RX_H_UID AND RXS.RX_D2_UID = BD.RX_D2_UID AND BD.ACTV_FLAG = 'Y' AND BD.STATUS = 'CFM')
INNER JOIN MM_BD_D1 BDI ON (BD.BD_H_UID = BDI.BD_H_UID AND RXI.RX_D1_UID = BDI.RX_D1_UID AND BDI.ACTV_FLAG = 'Y')
WHERE RX.ACTV_FLAG = 'Y'
AND RX.RFX_TYPE_CODE IN ('RQ', 'BD')
AND RXS.SEL_FLAG = 'Y'
AND BDI.SEL_FLAG = 'Y'
AND RX.STATUS = 'CFM' -- 업체선정
ORDER BY RX.RX_H_UID DESC;
